 16    DDataflow is an end2end tests solution.
 17    See our integrator manual for details.
 18    Additionally, use help(ddataflow) to see the available methods.

 21    _DEFAULT_SNAPSHOT_BASE_PATH = "dbfs:/ddataflow"
 22    _LOCAL_BASE_SNAPSHOT_PATH = os.environ["HOME"] + "/.ddataflow"
 23    _ENABLE_DDATAFLOW_ENVVARIABLE = "ENABLE_DDATAFLOW"
 24    _ENABLE_OFFLINE_MODE_ENVVARIABLE = "ENABLE_OFFLINE_MODE"
 25    _DDATAFLOW_CONFIG_FILE = "ddataflow_config.py"

 29    def __init__(
 30        self,
 31        project_folder_name: str,
 32        data_sources: Optional[dict] = None,
 33        data_writers: Optional[dict] = None,
 34        data_source_size_limit_gb: int = 1,
 35        enable_ddataflow=False,
 36        sources_with_default_sampling: Optional[List[str]] = None,
 37        snapshot_path: Optional[str] = None,
 38    ):
 39        """
 40        Initialize the dataflow object.
 41        The input of this object is the config dictionary outlined in our integrator manual.
 42
 43        Important params:
 44        project_folder_name:
 45            the name of the project that will be stored in the disk
 46        snapshot_path:
 47            path to the snapshot folder
 48        data_source_size_limit_gb:
 49            limit the size of the data sources
 50        sources_with_default_sampling:
 51         if you have tables you want to have by default and dont want to sample them first
 52        """
 53        self._size_limit = data_source_size_limit_gb
 54
 55        self.project_folder_name = project_folder_name
 56
 57        base_path = snapshot_path if snapshot_path else self._DEFAULT_SNAPSHOT_BASE_PATH
 58
 59        self._snapshot_path = base_path + "/" + project_folder_name
 60        self._local_path = self._LOCAL_BASE_SNAPSHOT_PATH + "/" + project_folder_name
 61
 62        if not data_sources:
 63            data_sources = {}
 64
 65        all_data_sources = {
 66            **build_default_sampling_for_sources(sources_with_default_sampling),
 67            **data_sources,
 68        }
 69
 70        self._data_sources = DataSources(
 71            config=all_data_sources,
 72            local_folder=self._local_path,
 73            snapshot_path=self._snapshot_path,
 74            size_limit=self._size_limit,
 75        )
 76
 77        self._data_writers: dict = data_writers if data_writers else {}
 78
 79        self._offline_enabled = os.getenv(self._ENABLE_OFFLINE_MODE_ENVVARIABLE, False)
 80
 81        self._ddataflow_enabled: Union[str, bool] = os.getenv(
 82            self._ENABLE_DDATAFLOW_ENVVARIABLE, enable_ddataflow
 83        )
 84
 85        # if offline is enabled we should use local data
 86        if self._offline_enabled:
 87            self.enable_offline()
 88
 89        self.save_sampled_data_sources = Sampler(
 90            self._snapshot_path, self._data_sources
 91        ).save_sampled_data_sources

100    def current_project() -> "DDataflow":
101        """
102        Returns a ddataflow configured with the current directory configuration file
103        Requirements for this to work:
104
105        1. MLTools must be called from withing the project root directory
106        2. There must be a file called ddataflow_config.py there
107        3. the module must have defined DDataflow object with the name of ddataflow
108
109        @todo investigate if we can use import_class_from_string
110        """
111        import sys
112
113        CONFIGURATION_FILE_NAME = "ddataflow_config.py"
114
115        current_folder = os.getcwd()
116        print("Loading config from folder", current_folder)
117        config_location = os.path.join(current_folder, CONFIGURATION_FILE_NAME)
118
119        if not os.path.exists(config_location):
120            raise Exception(
121                f"""
122This command needs to be executed within a project containing a {CONFIGURATION_FILE_NAME} file.
123You can start a new one for the current folder by running the following command:
124$ ddataflow setup_project"""
125            )
126
127        sys.path.append(current_folder)
128
129        import ddataflow_config
130
131        if hasattr(ddataflow_config, "ddataflow_client"):
132            return ddataflow_config.ddataflow_client
133
134        if not hasattr(ddataflow_config, "ddataflow"):
135            raise Exception("ddataflow object is not defined in your _config file")
136
137        return ddataflow_config.ddataflow

166        """
167        Gives access to the data source configured in the dataflow
168
169        You can also use this function in the terminal with --debugger=True to inspect the dataframe.
170        """

173
174        logger.info("Loading data source")
175        data_source: DataSource = self._data_sources.get_data_source(name)

177        df = self._get_df_from_source(data_source)

185    def source_name(self, name, disable_view_creation=False):
186        """
187        Given the name of a production table, returns the name of the corresponding ddataflow table when ddataflow is enabled
188        If ddataflow is disabled get the production one.
189        """
190        logger.info("Source name used: ", name)
191        source_name = name
192
193        if self._ddataflow_enabled:
194            source_name = self._get_new_table_name(name)
195            if disable_view_creation:
196                return source_name
197
198            print(f"Creating a temp view with the name: {source_name}")
199            data_source: DataSource = self._data_sources.get_data_source(name)
200
201            if self._offline_enabled:
202                df = data_source.query_locally(self._get_spark())
203            else:
204                df = data_source.query()
205
206            df.createOrReplaceTempView(source_name)
207
208            return source_name
209
210        return source_name

237    def _get_df_from_source(self, data_source):
238        if not self._ddataflow_enabled:
239            print("DDataflow not enabled")
240            # goes directly to production without prefilters
241            return data_source.query_without_filter()
242
243        if self._offline_enabled:
244            # uses snapshot data
245            if using_databricks_connect():
246                print(
247                    "Looks like you are using databricks-connect in offline mode. You probably want to run it "
248                    "without databricks connect in offline mode"
249                )
250
251            return data_source.query_locally()
252
253        print("DDataflow enabled and filtering")
254        return data_source.query()

275    def write(self, df, name: str):
276        """
277        Write a dataframe either to a local folder or the production one
278        """
279        if name not in self._data_writers:
280            raise WriterNotFoundException(name)
281
282        if self._ddataflow_enabled:
283            writing_path = self._snapshot_path
284
285            if self._offline_enabled:
286                writing_path = self._local_path
287            else:
288                if not writing_path.startswith(DDataflow._DEFAULT_SNAPSHOT_BASE_PATH):
289                    raise Exception(
290                        f"Only writing to {DDataflow._DEFAULT_SNAPSHOT_BASE_PATH} is enabled"
291                    )
292
293            writing_path = os.path.join(writing_path, name)
294            logger.info("Writing data to parquet file: " + writing_path)
295            return df.write.parquet(writing_path, mode="overwrite")
296
297        # if none of the above writes to production
298        return self._data_writers[name]["writer"](df, name, self._get_spark())  # type: ignore

301        """
302        Read the data writers parquet file which are stored in the ddataflow folder

304        path = self._snapshot_path
305        if self._offline_enabled:
306            path = self._local_path
307
308        parquet_path = os.path.join(path, name)
309        return self._get_spark().read.parquet(parquet_path)

357    def print_status(self):
358        """
359        Print the status of the ddataflow
360        """
361        if self._offline_enabled:
362            print("DDataflow is now ENABLED in OFFLINE mode")
363            print("To disable it remove from your code or unset the enviroment variable 'unset ENABLE_DDATAFLOW ; unset ENABLE_OFFLINE_MODE'")
364        elif self._ddataflow_enabled:
365            print(
366                """
367DDataflow is now ENABLED in ONLINE mode. Filtered data will be used and it will write to temporary tables.
368"""
369            )
370        else:
371            print(
372                f"""
373DDataflow is now DISABLED. So PRODUCTION data will be used and it will write to production tables.
374Use enable() function or export {self._ENABLE_DDATAFLOW_ENVVARIABLE}=True to enable it.
375If you are working offline use export ENABLE_OFFLINE_MODE=True instead.
376
377"""
378            )

380    def _get_current_environment_data_folder(self) -> Optional[str]:
381        if not self._ddataflow_enabled:
382            raise Exception("DDataflow is disabled so no data folder is available")
383
384        if self._offline_enabled:
385            return self._local_path
386
387        return self._snapshot_path
